Understanding Operators in C++

Operators are symbols that perform operations on variables and values.

Types of Operators in C++

C++ provides various operators categorized as follows:

Operator Type Description Example
Arithmetic Operators Perform mathematical operations +, -, *, /, %
Relational Operators Compare two values ==, !=, >, <, >=, <=
Logical Operators Used for logical operations &&, ||, !
Bitwise Operators Perform operations at the bit level &, |, ^, ~, <<, >>
Assignment Operators Assign values to variables =, +=, -=, *=, /=, %=
Increment & Decrement Increase or decrease value ++, --
Conditional (Ternary) Operator Shorthand for if-else condition ? expr1 : expr2

Example: Arithmetic Operators

#include <iostream>
using namespace std;

int main() {
    int a = 10, b = 5;
    
    cout << "Addition: " << (a + b) << endl;
    cout << "Subtraction: " << (a - b) << endl;
    cout << "Multiplication: " << (a * b) << endl;
    cout << "Division: " << (a / b) << endl;
    cout << "Modulus: " << (a % b) << endl;

    return 0;
}

Example: Relational Operators

#include <iostream>
using namespace std;

int main() {
    int x = 10, y = 20;
    
    cout << (x == y) << endl;
    cout << (x != y) << endl;
    cout << (x > y) << endl;
    cout << (x < y) << endl;

    return 0;
}

Example: Logical Operators

#include <iostream>
using namespace std;

int main() {
    bool a = true, b = false;
    
    cout << (a && b) << endl; // Logical AND
    cout << (a || b) << endl; // Logical OR
    cout << (!a) << endl;     // Logical NOT

    return 0;
}

Example: Bitwise Operators

#include <iostream>
using namespace std;

int main() {
    int x = 5, y = 3;
    
    cout << (x & y) << endl;  // Bitwise AND
    cout << (x | y) << endl;  // Bitwise OR
    cout << (x ^ y) << endl;  // Bitwise XOR
    cout << (~x) << endl;     // Bitwise NOT
    cout << (x << 1) << endl; // Left Shift
    cout << (y >> 1) << endl; // Right Shift

    return 0;
}

Example: Ternary Operator

#include <iostream>
using namespace std;

int main() {
    int a = 10, b = 20;
    int min = (a < b) ? a : b;

    cout << "Minimum value: " << min << endl;
    return 0;
}
